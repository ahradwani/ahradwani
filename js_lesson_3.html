// The function that bridges the HTML screen to our math logic
function processInputData() {
const consoleBox = document.getElementById('screen-console');
// 3. Reach out and grab the raw string value from the HTML input box
let rawInput = document.getElementById('user-number-box').value;
// Safety check: Did the user click the button without typing anything?
if (rawInput === "") {
consoleBox.innerText = "⚠️ Please type a number in the box first!";
return; // Stop running the function right here
}
// 4. CRITICAL: Convert the text string into an actual mathematical number
let cleanNumber = Number(rawInput);
// 5. Run the number through our machine
let finalResult = addOne(cleanNumber);
// 6. Print the beautiful math result back to the screen
consoleBox.innerText = "📥 You entered: " + cleanNumber + "\n" + "⚙️ Machine Output (Value + 1): " + finalResult;
}
