This simple code will use * to draw a diamond shape on the screen based on the
size the user will enter.
  # This line rows will be the diamond size.

rows = int(input("Enter number of rows: "))

# This code will draw the upper part of the diamond.

for i in range(0, rows):
    for j in range(0, rows - i -1):
        print(" ",end="")
    for j in range (0,i +1):
        print("* ",end="")
    print()
    
# This part will draw the second hulf of the diamond.

for i in range(0, rows -1):
    for j in range(0, i + 1 ):
        print(" ", end="")
    for j in range(0, rows -i - 1 ):
        print("* ", end="")
    print()
